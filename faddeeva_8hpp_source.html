29/* Available at: http://ab-initio.mit.edu/Faddeeva
30
31 Computes various error functions (erf, erfc, erfi, erfcx),
32 including the Dawson integral, in the complex plane, based
33 on algorithms for the computation of the Faddeeva function
34 w(z) = exp(-z^2) * erfc(-i*z).
35 Given w(z), the error functions are mostly straightforward
36 to compute, except for certain regions where we have to
37 switch to Taylor expansions to avoid cancellation errors
38 [e.g. near the origin for erf(z)].
39
40 To compute the Faddeeva function, we use a combination of two
41 algorithms:
42
43 For sufficiently large |z|, we use a continued-fraction expansion
44 for w(z) similar to those described in:
45
46 Walter Gautschi, "Efficient computation of the complex error
47 function," SIAM J. Numer. Anal. 7(1), pp. 187-198 (1970)
48
49 G. P. M. Poppe and C. M. J. Wijers, "More efficient computation
50 of the complex error function," ACM Trans. Math. Soft. 16(1),
51 pp. 38-46 (1990).
52
53 Unlike those papers, however, we switch to a completely different
54 algorithm for smaller |z|:
55
56 Mofreh R. Zaghloul and Ahmed N. Ali, "Algorithm 916: Computing the
57 Faddeyeva and Voigt Functions," ACM Trans. Math. Soft. 38(2), 15
58 (2011).
59
60 (I initially used this algorithm for all z, but it turned out to be
61 significantly slower than the continued-fraction expansion for
62 larger |z|. On the other hand, it is competitive for smaller |z|,
63 and is significantly more accurate than the Poppe & Wijers code
64 in some regions, e.g. in the vicinity of z=1+1i.)
65
66 Note that this is an INDEPENDENT RE-IMPLEMENTATION of these algorithms,
67 based on the description in the papers ONLY. In particular, I did
68 not refer to the authors' Fortran or Matlab implementations, respectively,
69 (which are under restrictive ACM copyright terms and therefore unusable
70 in free/open-source software).
71
72 Steven G. Johnson, Massachusetts Institute of Technology
73 http://math.mit.edu/~stevenj
74 October 2012.
75
76 -- Note that Algorithm 916 assumes that the erfc(x) function,
77 or rather the scaled function erfcx(x) = exp(x*x)*erfc(x),
78 is supplied for REAL arguments x. I originally used an
79 erfcx routine derived from DERFC in SLATEC, but I have
80 since replaced it with a much faster routine written by
81 me which uses a combination of continued-fraction expansions
82 and a lookup table of Chebyshev polynomials. For speed,
83 I implemented a similar algorithm for Im[w(x)] of real x,
84 since this comes up frequently in the other error functions.
85
86 A small test program is included the end, which checks
87 the w(z) etc. results against several known values. To compile
88 the test function, compile with -DTEST_FADDEEVA (that is,
89 #define TEST_FADDEEVA).
90
91 If HAVE_CONFIG_H is #defined (e.g. by compiling with -DHAVE_CONFIG_H),
92 then we #include "config.h", which is assumed to be a GNU autoconf-style
93 header defining HAVE_* macros to indicate the presence of features. In
94 particular, if HAVE_ISNAN and HAVE_ISINF are #defined, we use those
95 functions in math.h instead of defining our own, and if HAVE_ERF and/or
96 HAVE_ERFC are defined we use those functions from <cmath> for erf and
97 erfc of real arguments, respectively, instead of defining our own.

322// compute the error function erf(x)
323double FADDEEVA_RE(erf)(double x)
324{
325#if !defined(__cplusplus)
326 return erf(x); // C99 supplies erf in math.h
327#elif (__cplusplus >= 201103L) || defined(HAVE_ERF)
328 return ::erf(x); // C++11 supplies std::erf in cmath
329#else
330 double mx2 = -x*x;
331 if (mx2 < -750) // underflow
332 return (x >= 0 ? 1.0 : -1.0);
333
334 if (x >= 0) {
335 if (x < 8e-2) goto taylor;
336 return 1.0 - exp(mx2) * FADDEEVA_RE(erfcx)(x);
337 }
338 else { // x < 0
339 if (x > -8e-2) goto taylor;
340 return exp(mx2) * FADDEEVA_RE(erfcx)(-x) - 1.0;
341 }
342
343 // Use Taylor series for small |x|, to avoid cancellation inaccuracy
344 // erf(x) = 2/sqrt(pi) * x * (1 - x^2/3 + x^4/10 - x^6/42 + x^8/216 + ...)
345 taylor:
346 return x * (1.1283791670955125739
347 + mx2 * (0.37612638903183752464
348 + mx2 * (0.11283791670955125739
349 + mx2 * (0.026866170645131251760
350 + mx2 * 0.0052239776254421878422))));
351#endif
352}

354// compute the error function erf(z)
355cmplx FADDEEVA(erf)(cmplx z, double relerr)
356{
357 double x = creal(z), y = cimag(z);
358
359 if (y == 0)
360 return C(FADDEEVA_RE(erf)(x),
361 y); // preserve sign of 0
362 if (x == 0) // handle separately for speed & handling of y = Inf or NaN
363 return C(x, // preserve sign of 0
364 /* handle y -> Inf limit manually, since
365 exp(y^2) -> Inf but Im[w(y)] -> 0, so
366 IEEE will give us a NaN when it should be Inf */
367 y*y > 720 ? (y > 0 ? Inf : -Inf)
368 : exp(y*y) * FADDEEVA(w_im)(y));
369
370 double mRe_z2 = (y - x) * (x + y); // Re(-z^2), being careful of overflow
371 double mIm_z2 = -2*x*y; // Im(-z^2)
372 if (mRe_z2 < -750) // underflow
373 return (x >= 0 ? 1.0 : -1.0);
374
375 /* Handle positive and negative x via different formulas,
376 using the mirror symmetries of w, to avoid overflow/underflow
377 problems from multiplying exponentially large and small quantities. */
378 if (x >= 0) {
379 if (x < 8e-2) {
380 if (fabs(y) < 1e-2)
381 goto taylor;
382 else if (fabs(mIm_z2) < 5e-3 && x < 5e-3)
383 goto taylor_erfi;
384 }
385 /* don't use complex exp function, since that will produce spurious NaN
386 values when multiplying w in an overflow situation. */
387 return 1.0 - exp(mRe_z2) *
388 (C(cos(mIm_z2), sin(mIm_z2))
389 * FADDEEVA(w)(C(-y,x), relerr));
390 }
391 else { // x < 0
392 if (x > -8e-2) { // duplicate from above to avoid fabs(x) call
393 if (fabs(y) < 1e-2)
394 goto taylor;
395 else if (fabs(mIm_z2) < 5e-3 && x > -5e-3)
396 goto taylor_erfi;
397 }
398 else if (isnan(x))
399 return C(NaN, y == 0 ? 0 : NaN);
400 /* don't use complex exp function, since that will produce spurious NaN
401 values when multiplying w in an overflow situation. */
402 return exp(mRe_z2) *
403 (C(cos(mIm_z2), sin(mIm_z2))
404 * FADDEEVA(w)(C(y,-x), relerr)) - 1.0;
405 }
406
407 // Use Taylor series for small |z|, to avoid cancellation inaccuracy
408 // erf(z) = 2/sqrt(pi) * z * (1 - z^2/3 + z^4/10 - z^6/42 + z^8/216 + ...)
409 taylor:
410 {
411 cmplx mz2 = C(mRe_z2, mIm_z2); // -z^2
412 return z * (1.1283791670955125739
413 + mz2 * (0.37612638903183752464
414 + mz2 * (0.11283791670955125739
415 + mz2 * (0.026866170645131251760
416 + mz2 * 0.0052239776254421878422))));
417 }
418
419 /* for small |x| and small |xy|,
420 use Taylor series to avoid cancellation inaccuracy:
421 erf(x+iy) = erf(iy)
422 + 2*exp(y^2)/sqrt(pi) *
423 [ x * (1 - x^2 * (1+2y^2)/3 + x^4 * (3+12y^2+4y^4)/30 + ...
424 - i * x^2 * y * (1 - x^2 * (3+2y^2)/6 + ...) ]
425 where:
426 erf(iy) = exp(y^2) * Im[w(y)]
427 */
428 taylor_erfi:
429 {
430 double x2 = x*x, y2 = y*y;
431 double expy2 = exp(y2);
432 return C
433 (expy2 * x * (1.1283791670955125739
434 - x2 * (0.37612638903183752464
435 + 0.75225277806367504925*y2)
436 + x2*x2 * (0.11283791670955125739
437 + y2 * (0.45135166683820502956
438 + 0.15045055561273500986*y2))),
439 expy2 * (FADDEEVA(w_im)(y)
440 - x2*y * (1.1283791670955125739
441 - x2 * (0.56418958354775628695
442 + 0.37612638903183752464*y2))));
443 }
444}

475// erfc(z) = 1 - erf(z)
476cmplx FADDEEVA(erfc)(cmplx z, double relerr)
477{
478 double x = creal(z), y = cimag(z);
479
480 if (x == 0.)
481 return C(1,
482 /* handle y -> Inf limit manually, since
483 exp(y^2) -> Inf but Im[w(y)] -> 0, so
484 IEEE will give us a NaN when it should be Inf */
485 y*y > 720 ? (y > 0 ? -Inf : Inf)
486 : -exp(y*y) * FADDEEVA(w_im)(y));
487 if (y == 0.) {
488 if (x*x > 750) // underflow
489 return C(x >= 0 ? 0.0 : 2.0,
490 -y); // preserve sign of 0
491 return C(x >= 0 ? exp(-x*x) * FADDEEVA_RE(erfcx)(x)
492 : 2. - exp(-x*x) * FADDEEVA_RE(erfcx)(-x),
493 -y); // preserve sign of zero
494 }
495
496 double mRe_z2 = (y - x) * (x + y); // Re(-z^2), being careful of overflow
497 double mIm_z2 = -2*x*y; // Im(-z^2)
498 if (mRe_z2 < -750) // underflow
499 return (x >= 0 ? 0.0 : 2.0);
500
501 if (x >= 0)
502 return cexp(C(mRe_z2, mIm_z2))
503 * FADDEEVA(w)(C(-y,x), relerr);
504 else
505 return 2.0 - cexp(C(mRe_z2, mIm_z2))
506 * FADDEEVA(w)(C(y,-x), relerr);
507}

510double FADDEEVA_RE(Dawson)(double x)

516// compute Dawson(z) = sqrt(pi)/2 * exp(-z^2) * erfi(z)
517cmplx FADDEEVA(Dawson)(cmplx z, double relerr)
518{
519 const double spi2 = 0.8862269254527580136490837416705725913990; // sqrt(pi)/2
520 double x = creal(z), y = cimag(z);
521
522 // handle axes separately for speed & proper handling of x or y = Inf or NaN
523 if (y == 0)
524 return C(spi2 * FADDEEVA(w_im)(x),
525 -y); // preserve sign of 0
526 if (x == 0) {
527 double y2 = y*y;
528 if (y2 < 2.5e-5) { // Taylor expansion
529 return C(x, // preserve sign of 0
530 y * (1.
531 + y2 * (0.6666666666666666666666666666666666666667
532 + y2 * 0.26666666666666666666666666666666666667)));
533 }
534 return C(x, // preserve sign of 0
535 spi2 * (y >= 0
536 ? exp(y2) - FADDEEVA_RE(erfcx)(y)
537 : FADDEEVA_RE(erfcx)(-y) - exp(y2)));
538 }
539
540 double mRe_z2 = (y - x) * (x + y); // Re(-z^2), being careful of overflow
541 double mIm_z2 = -2*x*y; // Im(-z^2)
542 cmplx mz2 = C(mRe_z2, mIm_z2); // -z^2
543
544 /* Handle positive and negative x via different formulas,
545 using the mirror symmetries of w, to avoid overflow/underflow
546 problems from multiplying exponentially large and small quantities. */
547 if (y >= 0) {
548 if (y < 5e-3) {
549 if (fabs(x) < 5e-3)
550 goto taylor;
551 else if (fabs(mIm_z2) < 5e-3)
552 goto taylor_realaxis;
553 }
554 cmplx res = cexp(mz2) - FADDEEVA(w)(z, relerr);
555 return spi2 * C(-cimag(res), creal(res));
556 }
557 else { // y < 0
558 if (y > -5e-3) { // duplicate from above to avoid fabs(x) call
559 if (fabs(x) < 5e-3)
560 goto taylor;
561 else if (fabs(mIm_z2) < 5e-3)
562 goto taylor_realaxis;
563 }
564 else if (isnan(y))
565 return C(x == 0 ? 0 : NaN, NaN);
566 cmplx res = FADDEEVA(w)(-z, relerr) - cexp(mz2);
567 return spi2 * C(-cimag(res), creal(res));
568 }

578 use Taylor series to avoid cancellation inaccuracy:
579 dawson(x + iy)

583 + y^4/15 (4 - 15Dx - 9x^2 + 20Dx^3 + 2x^4 - 4Dx^5) ] + ...
584 where D = dawson(x)
585

587 this series (many of the leading terms cancel). So, for large |x|,
588 we need to substitute a continued-fraction expansion for D.
589
590 dawson(x) = 0.5 / (x-0.5/(x-1/(x-1.5/(x-2/(x-2.5/(x...))))))

593 accurate as soon as the simpler Taylor expansion above starts

603 expansion for the real part, and a 2-term expansion for the imaginary
604 part. (This avoids overflow problems for huge |x|.) This yields:
605
606 Re dawson(x + iy) = [1 + y^2 (1 + y^2/2 - (xy)^2/3)] / (2x)
607 Im dawson(x + iy) = y [ -1 - 2/3 y^2 + y^4/15 (2x^2 - 4) ] / (2x^2 - 1)

610 taylor_realaxis:

612 double x2 = x*x;
613 if (x2 > 1600) { // |x| > 40
614 double y2 = y*y;
615 if (x2 > 25e14) {// |x| > 5e7
616 double xy2 = (x*y)*(x*y);
617 return C((0.5 + y2 * (0.5 + 0.25*y2
618 - 0.16666666666666666667*xy2)) / x,
619 y * (-1 + y2 * (-0.66666666666666666667
620 + 0.13333333333333333333*xy2
621 - 0.26666666666666666667*y2))
622 / (2*x2 - 1));
623 }
624 return (1. / (-15 + x2*(90 + x2*(-60 + 8*x2)))) *
625 C(x * (33 + x2 * (-28 + 4*x2)
626 + y2 * (18 - 4*x2 + 4*y2)),
627 y * (-15 + x2 * (24 - 4*x2)
628 + y2 * (4*x2 - 10 - 4*y2)));
629 }
630 else {
631 double D = spi2 * FADDEEVA(w_im)(x);
632 double y2 = y*y;
633 return C
634 (D + y2 * (D + x - 2*D*x2)
635 + y2*y2 * (D * (0.5 - x2 * (2 - 0.66666666666666666667*x2))
636 + x * (0.83333333333333333333
637 - 0.33333333333333333333 * x2)),
638 y * (1 - 2*D*x
639 + y2 * 0.66666666666666666667 * (1 - x2 - D*x * (3 - 2*x2))
640 + y2*y2 * (0.26666666666666666667 -
641 x2 * (0.6 - 0.13333333333333333333 * x2)
642 - D*x * (1 - x2 * (1.3333333333333333333
643 - 0.26666666666666666667 * x2)))));
644 }
645 }

854 where we compute all of the sums is faster (at least with the
855 precomputed expa2n2 table) until about x=10. Furthermore, if we
856 try to compute all of the sums for x > 20, I find that we
857 sometimes run into numerical problems because underflow/overflow

859 below. Therefore, we use x < 10 here. */

988 double n0 = floor(x/a + 0.5); // sum in both directions, starting at n0
989 double dx = a*n0 - x;
990 sum3 = exp(-dx*dx) / (a2*(n0*n0) + y*y);
991 sum5 = a*n0 * sum3;
992 double exp1 = exp(4*a*dx), exp1dn = 1;
993 int dn;
994 for (dn = 1; n0 - dn > 0; ++dn) { // loop over n0-dn and n0+dn terms
995 double np = n0 + dn, nm = n0 - dn;
996 double tp = exp(-sqr(a*dn+dx));
997 double tm = tp * (exp1dn *= exp1); // trick to get tm from tp
998 tp /= (a2*(np*np) + y*y);
999 tm /= (a2*(nm*nm) + y*y);
1000 sum3 += tp + tm;
1001 sum5 += a * (np * tp + nm * tm);
1002 if (a * (np * tp + nm * tm) < relerr * sum5) goto finish;
1003 }
1004 while (1) { // loop over n0+dn terms only (since n0-dn <= 0)
1005 double np = n0 + dn++;
1006 double tp = exp(-sqr(a*dn+dx)) / (a2*(np*np) + y*y);
1007 sum3 += tp;
1008 sum5 += a * np * tp;
1009 if (a * np * tp < relerr * sum5) goto finish;
1010 }
1011 }
1012 finish:
1013 return ret + C((0.5*c)*y*(sum2+sum3),
1014 (0.5*c)*copysign(sum5-sum4, creal(z)));

1019/* erfcx(x) = exp(x^2) erfc(x) function, for real x, written by
1020 Steven G. Johnson, October 2012.
1021
1022 This function combines a few different ideas.
1023
1024 First, for x > 50, it uses a continued-fraction expansion (same as
1025 for the Faddeeva function, but with algebraic simplifications for z=i*x).
1026
1027 Second, for 0 <= x <= 50, it uses Chebyshev polynomial approximations,
1028 but with two twists:
1029
1030 a) It maps x to y = 4 / (4+x) in [0,1]. This simple transformation,
1031 inspired by a similar transformation in the octave-forge/specfun
1032 erfcx by Soren Hauberg, results in much faster Chebyshev convergence
1033 than other simple transformations I have examined.
1034
1035 b) Instead of using a single Chebyshev polynomial for the entire
1036 [0,1] y interval, we break the interval up into 100 equal
1037 subintervals, with a switch/lookup table, and use much lower
1038 degree Chebyshev polynomials in each subinterval. This greatly
1039 improves performance in my tests.
1040
1041 For x < 0, we use the relationship erfcx(-x) = 2 exp(x^2) - erfc(x),
1042 with the usual checks for overflow etcetera.
1043
1044 Performance-wise, it seems to be substantially faster than either
1045 the SLATEC DERFC function [or an erfcx function derived therefrom]
1046 or Cody's CALERF function (from netlib.org/specfun), while
1047 retaining near machine precision in accuracy. */

1049/* Given y100=100*y, where y = 4/(4+x) for x >= 0, compute erfc(x).
1050
1051 Uses a look-up table of 100 different Chebyshev polynomials
1052 for y intervals [0,0.01], [0.01,0.02], ...., [0.99,1], generated
1053 with the help of Maple and a little shell script. This allows
1054 the Chebyshev polynomials to be of significantly lower degree (about 1/4)
1055 compared to fitting the whole [0,1] interval with a single polynomial. */
1056static double erfcx_y100(double y100)
1057{
1058 switch ((int) y100) {
1059case 0: {
1060double t = 2*y100 - 1;
1061return 0.70878032454106438663e-3 + (0.71234091047026302958e-3 + (0.35779077297597742384e-5 + (0.17403143962587937815e-7 + (0.81710660047307788845e-10 + (0.36885022360434957634e-12 + 0.15917038551111111111e-14 * t) * t) * t) * t) * t) * t;
1062}

1944// compute relative error |b-a|/|a|, handling case of NaN and Inf,
1945static double relerr(double a, double b) {
1946 if (isnan(a) || isnan(b) || isinf(a) || isinf(b)) {
1947 if ((isnan(a) && !isnan(b)) || (!isnan(a) && isnan(b)) ||
1948 (isinf(a) && !isinf(b)) || (!isinf(a) && isinf(b)) ||
1949 (isinf(a) && isinf(b) && a*b < 0))
1950 return Inf; // "infinite" error
1951 return 0; // matching infinity/nan results counted as zero error
1952 }
1953 if (a == 0)
1954 return b == 0 ? 0 : Inf;
1955 else
1956 return fabs((b-a) / a);
1957}
