23template <typename T> T custom(T x, T alpha, T beta, T sigma, T x_max, T N) {
24 T y = x / x_max;
25 return y < 0.0 or y > 1.0 ? 0.0
26 : N * std::pow(y, alpha) * std::pow(1.0 - y, beta) *
27 std::exp(-std::pow((x - x_max) / sigma, 2));
