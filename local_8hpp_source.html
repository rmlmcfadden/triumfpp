42template <typename T = double>
43T slr_rate_z(T z, T temperature, T critical_temperature, T lambda_0, T exponent,
44 T applied_field, T dipole_field, T correlation_rate,
45 T slr_constant, T slr_exponent, T surface_thickness,
46 T surface_rate) {
47 // correct depth for the surface layer
48 T _z_ = z - surface_thickness;
49 if (_z_ < 0.0) {
50 return surface_rate;
51 } else {
52 // calculate the local field from the screening profile
53 T lambda = triumf::superconductivity::phenomenology::penetration_depth(
54 temperature, critical_temperature, exponent, lambda_0);
55 T screened_field = temperature > critical_temperature
56 ? applied_field
57 : applied_field * std::exp(-1.0 * _z_ / lambda);
58 // calculate the dipole-dipole SLR rate in the superconducting state
59 T dd_rate = triumf::nmr::dipole_dipole::slr_rate<T>(
60 screened_field, dipole_field, correlation_rate,
61 triumf::bnmr::nuclei::lithium_8<T>::gyromagnetic_ratio(),
62 triumf::nmr::nuclei::niobium_93<T>::gyromagnetic_ratio());
63 // calculate the SLR rate in the normal state
64 T ns_rate = slr_constant * std::pow(temperature, slr_exponent);
65 // return the "surface" contribution at shallow depths
66 return dd_rate + ns_rate;
67 }
