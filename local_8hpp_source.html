44template <typename T = double>
45T slr_rate_z(T z, T temperature, T critical_temperature, T lambda_0, T exponent,
46 T applied_field, T dipole_field, T correlation_rate,
47 T slr_constant, T slr_exponent, T surface_thickness,
48 T surface_rate) {
49 // correct for the field-dependence to the critical temperature
50 // https://doi.org/10.1103/PhysRevB.2.3545
51 // https://doi.org/10.1016/j.nima.2004.09.003
52 // https://doi.org/10.1088/0953-2048/25/6/065014
53 constexpr T Nb_B_c2 = 0.425; // T
54 T corrected_critical_temperature =
55 triumf::superconductivity::phenomenology::critical_temperature<T>(
56 applied_field, critical_temperature, Nb_B_c2, 0.5);
57 // correct depth for the surface layer
58 T _z_ = z - surface_thickness;
59 if (_z_ < 0.0) {
60 return surface_rate;
61 } else {
62 // calculate the local field from the screening profile
63 T lambda = triumf::superconductivity::phenomenology::penetration_depth(
64 temperature, corrected_critical_temperature, exponent, lambda_0);
65 T screened_field = temperature > corrected_critical_temperature
66 ? applied_field
67 : applied_field * std::exp(-1.0 * _z_ / lambda);
68 // calculate the dipole-dipole SLR rate in the superconducting state
69 T dd_rate = triumf::nmr::dipole_dipole::slr_rate<T>(
70 screened_field, dipole_field, correlation_rate,
71 triumf::bnmr::nuclei::lithium_8<T>::gyromagnetic_ratio(),
72 triumf::nmr::nuclei::niobium_93<T>::gyromagnetic_ratio());
73 // calculate the SLR rate in the normal state
74 T ns_rate = slr_constant * std::pow(temperature, slr_exponent);
75 // return the "surface" contribution at shallow depths
76 return dd_rate + ns_rate;
77 }
78}

128template <typename T = double>
129T slr_rate_film_z(T z, T temperature, T critical_temperature, T lambda_0,
130 T exponent, T applied_field, T dipole_field,
131 T correlation_rate, T slr_constant, T slr_exponent,
132 T surface_thickness, T surface_rate, T film_thickness) {
133 // correct for the field-dependence to the critical temperature
134 // https://doi.org/10.1103/PhysRevB.2.3545
135 // https://doi.org/10.1016/j.nima.2004.09.003
136 // https://doi.org/10.1088/0953-2048/25/6/065014
137 constexpr T Nb_B_c2 = 0.425; // T
138 T corrected_critical_temperature =
139 triumf::superconductivity::phenomenology::critical_temperature<T>(
140 applied_field, critical_temperature, Nb_B_c2, 0.5);
141 // correct depth for the surface layer
142 T _z_ = z - surface_thickness;
143 // correct film thickness for the surface layer
144 T _d_ = 0.5 * (film_thickness - surface_thickness);
145 if (_z_ < 0.0) {
146 return surface_rate;
147 } else {
148 // calculate the local field from the screening profile
149 T lambda = triumf::superconductivity::phenomenology::penetration_depth(
150 temperature, corrected_critical_temperature, exponent, lambda_0);
151 T screened_field = temperature > corrected_critical_temperature
152 ? applied_field
153 : applied_field *
154 std::cosh((0.5 * _d_ - _z_) / lambda) /
155 std::cosh(0.5 * _d_ / lambda);
156 // calculate the dipole-dipole SLR rate in the superconducting state
157 T dd_rate = triumf::nmr::dipole_dipole::slr_rate<T>(
158 screened_field, dipole_field, correlation_rate,
159 triumf::bnmr::nuclei::lithium_8<T>::gyromagnetic_ratio(),
160 triumf::nmr::nuclei::niobium_93<T>::gyromagnetic_ratio());
161 // calculate the SLR rate in the normal state
162 T ns_rate = slr_constant * std::pow(temperature, slr_exponent);
163 // return the "surface" contribution at shallow depths
164 return dd_rate + ns_rate;
165 }
