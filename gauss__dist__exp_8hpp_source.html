20template <typename T = double>
21T pulsed_gauss_dist_exp_integral(T time, T time_p, T nuclear_lifetime,
22 T slr_rate, T sigma) {
23 // make sure that
24 assert(time >= time_p);
25 // integrand for the numeric integral
26 auto integrand = [=](T t_p) {
27 // translated from Maxima
28 T expo1 = std::exp(
29 boost::math::constants::half<T>() * time * time * sigma * sigma +
30 boost::math::constants::half<T>() * t_p * t_p * sigma * sigma +
31 t_p * slr_rate);
32 T expo2 = std::exp(-time / nuclear_lifetime + t_p / nuclear_lifetime -
33 t_p * time * sigma * sigma);
34 T expo3 = std::exp(time * slr_rate);
35 return -expo2 *
36 (expo1 * std::erf(((time - t_p) * sigma * sigma - slr_rate) /
37 boost::math::constants::root_two<T>() / sigma) -
38 expo1) /
39 (expo3 * std::erf(slr_rate / sigma /
40 boost::math::constants::root_two<T>()) +
41 expo3);
42 };
43 // create the integrator for tanh-sinh quadrature
44 static boost::math::quadrature::tanh_sinh<T> integrator;
45 // evaluate the integral from 0 to time_p
46 T Q = integrator.integrate(integrand, 0.0, time_p);
47 return Q;
48}

51template <typename T = double>
52T pulsed_gauss_dist_exp(T time, T nuclear_lifetime, T pulse_length, T asymmetry,
53 T slr_rate, T sigma) {
54 if (time == 0.0) {
55 return asymmetry;
56 } else if (time > 0.0 and time <= pulse_length) {
57 return asymmetry *
58 pulsed_gauss_dist_exp_integral(time, time, nuclear_lifetime,
59 slr_rate, sigma) /
60 normalization(time, nuclear_lifetime);
61 } else if (time > pulse_length) {
62 return (asymmetry *
63 pulsed_gauss_dist_exp_integral(time, pulse_length, nuclear_lifetime,
64 slr_rate, sigma) /
65 normalization(pulse_length, nuclear_lifetime)) /
66 std::exp(-(time - pulse_length) / nuclear_lifetime);
67 } else {
68 return 0.0;
69 }
70}
