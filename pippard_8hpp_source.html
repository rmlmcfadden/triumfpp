22template <typename T = double>
23T J_0(T temperature, T critical_temperature, T gap_meV, T exponent) {
24 // Boltzmann constant (meV / K)
25 constexpr T k_B_meV_per_K =
26 1e3 *
27 triumf::constants::codata_2018::Boltzmann_constant_in_eV_K<T>::value();
28 //
29 // T reduced_temperature = temperature / critical_temperature;
30 //
31 return std::pow(phenomenology::reduced_penetration_depth<T>(
32 temperature, critical_temperature, exponent),
33 2) *
34 bcs::reduced_gap<T>(temperature, critical_temperature) *
35 std::tanh(bcs::gap<T>(temperature, critical_temperature, gap_meV) /
36 (2.0 * k_B_meV_per_K * temperature));
37}

41template <typename T = double>
42T coherence_length(T temperature, T critical_temperature, T gap_meV, T exponent,
43 T xi_0, T mean_free_path) {
44 T fraction_1 =
45 J_0<T>(temperature, critical_temperature, gap_meV, exponent) / xi_0;
46 T fraction_2 = 1.0 / mean_free_path;
47 T fraction = fraction_1 + fraction_2;
48 // handle a some edge cases for very big/small args
49 if (std::isinf(fraction)) {
50 return 0.0;
51 } else if (fraction == 0.0) {
52 return std::numeric_limits<T>::infinity();
53 } else {
54 return 1.0 / fraction;
55 }
56}

70template <typename T = double> T g(T x) {
71 /*
72 boost::multiprecision::cpp_dec_float_100 value =
73 (3.0 / 2.0) * ((1.0 + x * x) * std::atan(x) - x) / (x * x * x);
74 return value;
75 */
76 // return correct result when x ~ 0.
77 if (x < 1.0e-4) {
78 return 1.0;
79 } else {
80 return (3.0 / 2.0) * ((1.0 + x * x) * std::atan(x) - x) / (x * x * x);

85template <typename T = double>
86T kernel(T q, T temperature, T critical_temperature, T gap_meV, T xi_0,
87 T mean_free_path, T lambda_0, T exponent) {
88 T x = q * coherence_length<T>(temperature, critical_temperature, gap_meV,
89 exponent, xi_0, mean_free_path);
90 return std::pow(phenomenology::penetration_depth<T>(
91 temperature, critical_temperature, exponent, lambda_0),
92 -2) *
93 reduced_coherence_length<T>(temperature, critical_temperature, gap_meV,
94 exponent, xi_0, mean_free_path) *
95 g<T>(x);
96}

109template <typename T = double>
110T reduced_field_penetration(T z, T temperature, T critical_temperature,
111 T gap_meV, T xi_0, T mean_free_path, T lambda_0,
112 T exponent) {
113 //
114 if (z <= 0.0) {
115 return 1.0;
116 } else {
117 //
118 auto pippard_integrand = [&](T q) -> T {
119 T K = kernel<T>(q, temperature, critical_temperature, gap_meV, xi_0,
120 mean_free_path, lambda_0, exponent);
121 return q / (q * q + K);
122 };
123 // create the integrator w/ specific tolerance and evaluation levels
124 // (defaults: root_epsilon and eight levels for type double).
125 // static T tolerance = std::sqrt(std::numeric_limits<T>::epsilon());
126 static T tolerance = std::cbrt(std::numeric_limits<T>::epsilon());
127 std::size_t levels = sizeof(T);
128 static boost::math::quadrature::ooura_fourier_sin<T> pippard_integrator =
129 boost::math::quadrature::ooura_fourier_sin<T>(tolerance, levels);
130 // evaluate the integral, which returns a pair
131 // (first = integral, second = relative error)
132 std::pair<T, T> result = pippard_integrator.integrate(pippard_integrand, z);
133 // return the integral multiplied by the prefactors to get B vs. z
134 return boost::math::constants::two_div_pi<T>() * result.first;
135 }
136}
