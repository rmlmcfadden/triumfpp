25template <typename T = double> T reduced_gap_solver(T x) {
26 // Use 1 as the initial guess for the Delta(t).
27 // This ensures that the first "root" is found as Delta(t) -> 0.0,
28 // otherwise the wrong solution is found when t >= 0.9.
29 T guess = 1.0;
30 // T guess = std::cos(boost::math::constants::half_pi<T>() * std::pow(x, 2));
31
32 // Bound the possible values for the gap.
33 // The adjustments by machine epsilon needed for "exact" results!?
34 T min = 0.0 - std::numeric_limits<T>::epsilon();
35 T max = 1.0 + std::numeric_limits<T>::epsilon();
36
37 // Maximum possible binary digits accuracy for type T.
38 const int digits = std::numeric_limits<T>::digits;
39
40 // Digits used to control how accurate to try to make the result.
41 // Accuracy triples with each step, so stop when just over one third of the
42 // digits are correct.
43 int get_digits = static_cast<int>(0.4 * digits);
44
45 // Limit the number of iterations taken to find the "root" of the expression.
46 // boost::uintmax_t max_iterations = 20;
47 boost::uintmax_t max_iterations =
48 std::numeric_limits<boost::uintmax_t>::max();
49
50 // Return the "root" of Thouless' Eqn. using both its 1st and 2nd derivatives.
51 T result = boost::math::tools::halley_iterate(
52 [x](const T &delta) {
53 // Return a tuple containing: f(delta), f'(delta), and f''(delta).
54 T f = std::tanh(delta / x) - delta;
55 T df_dx = 1.0 / (x * std::pow(std::cosh(delta / x), 2)) - 1.0;
56 T df2_dx2 =
57 -2.0 * std::tanh(delta / x) / std::pow(x * std::cosh(delta / x), 2);
58 return std::make_tuple(f, df_dx, df2_dx2);
59 },
60 guess, min, max, get_digits, max_iterations);
61 //
62 return result;
63}

130template <typename T = double>
131T coherence_length(T temperature, T critical_temperature, T gap_meV, T xi_0,
132 T mean_free_path, T n) {
133 T fraction_1 = boost::math::constants::two_div_pi<T>() *
134 f<T>(temperature, critical_temperature, gap_meV, n) *
135 reduced_gap<T>(temperature, critical_temperature) / xi_0;
136 T fraction_2 = 1.0 / mean_free_path;
137 T fraction = fraction_1 + fraction_2;
138 // handle a some edge cases for very big/small args
139 if (std::isinf(fraction)) {
140 return 0.0;
141 } else if (fraction == 0.0) {
142 return std::numeric_limits<T>::infinity();
143 } else {
144 return 1.0 / fraction;
145 }
146}

149template <typename T = double>
150T Lambda(T temperature, T critical_temperature, T gap_meV, T xi_0,
151 T mean_free_path, T lambda_0, T exponent, T n) {
152 return std::pow(
153 lambda_0,
154 // triumf::superconductivity::phenomenology::penetration_depth<T>(
155 // temperature, critical_temperature, exponent, lambda_0),
156 2) *
157 std::pow(f<T>(temperature, critical_temperature, gap_meV, n), 3) *

159 xi_0, mean_free_path, n) /
160 mean_free_path) /

165template <typename T = double> T g(T x) {
166 // return correct result when x ~ 0.
167 if (x < 1.0e-4) {
168 return 1.0;
169 } else {
170 return (3.0 / 2.0) * ((1.0 + x * x) * std::atan(x) - x) / (x * x * x);
171 }

175template <typename T = double>
176T kernel(T q, T temperature, T critical_temperature, T gap_meV, T xi_0,
177 T mean_free_path, T lambda_0, T exponent) {
178 //
179 T sum = 0.0;
180 T change = 0.0;
181 // const T precision = std::sqrt(std::numeric_limits<T>::epsilon());
182 const T precision = std::numeric_limits<T>::epsilon();
183 const T max_iterations = 100;
184 T n = 0.0;
185 do {
186 //
187 T x = q * coherence_length<T>(temperature, critical_temperature, gap_meV,
188 xi_0, mean_free_path, n);
189 //
190 change = g<T>(x) / Lambda<T>(temperature, critical_temperature, gap_meV,
191 xi_0, mean_free_path, lambda_0, exponent, n);
192 //
193 sum += change;
194 n += 1.0;
195 } while ((std::abs(change) > precision) and (n < max_iterations));
196 //
197 return sum;
198}

220template <typename T = double>
221T reduced_field_penetration(T z, T temperature, T critical_temperature,
222 T gap_meV, T xi_0, T mean_free_path, T lambda_0,
223 T exponent) {
224 //
225 if (z <= 0.0) {
226 return 1.0;
227 } else {
228 //
229 auto bcs_integrand = [&](T q) -> T {
230 T K = kernel<T>(q, temperature, critical_temperature, gap_meV, xi_0,
231 mean_free_path, lambda_0, exponent);
232 return q / (q * q + K);
233 };
234 // create the integrator w/ default tolerance and evaluation levels
235 // (root_epsilon and eight levels for type double).
236 static boost::math::quadrature::ooura_fourier_sin<T> bcs_integrator =
237 boost::math::quadrature::ooura_fourier_sin<T>();
238 // evaluate the integral, which returns a pair
239 // (first = integral, second = relative error)
240 std::pair<T, T> result = bcs_integrator.integrate(bcs_integrand, z);
241 // return the integral multiplied by the prefactors to get B vs. z
242 return boost::math::constants::two_div_pi<T>() * result.first;
243 }
244}
