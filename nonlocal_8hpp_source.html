42template <typename T = double>
43T slr_rate_z(T z, T temperature, T critical_temperature, T gap_meV, T xi_0,
44 T mean_free_path, T lambda_0, T exponent, T applied_field,
45 T dipole_field, T correlation_rate, T slr_constant, T slr_exponent,
46 T surface_thickness, T surface_rate, T electron_phonon_coupling) {
47 // correct depth for the surface layer
48 T _z_ = z - surface_thickness;
49 if (_z_ < 0.0) {
50 return surface_rate;
51 } else {
52 // convert to values in the weak coupling limit
53 // see e.g., Eqs. (1) & (2) in:
54 // http://dx.doi.org/10.1103/PhysRevB.87.104508
55 T lambda_0_wc = xi_0 * electron_phonon_coupling;
56 T xi_0_wc = lambda_0 / std::sqrt(electron_phonon_coupling);
57 // calculate the local field from the screening profile
58 T screened_field =
59 temperature > critical_temperature
60 ? applied_field
61 : triumf::superconductivity::pippard::field_penetration<T>(
62 _z_, temperature, critical_temperature, gap_meV, xi_0_wc,
63 mean_free_path, lambda_0_wc, exponent, applied_field);
64 // calculate the dipole-dipole SLR rate in the superconducting state
65 T dd_rate = triumf::nmr::dipole_dipole::slr_rate<T>(
66 screened_field, dipole_field, correlation_rate,
67 triumf::bnmr::nuclei::lithium_8<T>::gyromagnetic_ratio(),
68 triumf::nmr::nuclei::niobium_93<T>::gyromagnetic_ratio());
69 // calculate the SLR rate in the normal state
70 T ns_rate = slr_constant * std::pow(temperature, slr_exponent);
71 // return the "surface" contribution at shallow depths
72 return dd_rate + ns_rate;
73 }

171 // depth-averaging using "histogram" summation
172 T operator()(T energy_keV) {
173 // bin edges
174 std::vector<T> z_edge =
175 triumf::numpy::linspace<T>(0.0, z_max(energy_keV), n_bins);
176 // bin widths - adjust ranges by one for "correct" size
177 std::vector<T> dz(n_bins - 1);
178 std::adjacent_difference(std::begin(z_edge) + 1, std::end(z_edge),
179 std::begin(dz));
180 // bin centres
181 std::vector<T> z(n_bins - 1);
182 // probabilities
183 std::vector<T> p_z(n_bins - 1);
184 std::vector<T> weights(n_bins - 1);
185 std::vector<T> slr_rates(n_bins - 1);
186
187 for (std::size_t i = 0; i < z.size(); ++i) {
188 z.at(i) = z_edge.at(i) + 0.5 * dz.at(i);
189 p_z.at(i) = triumf::srim::pdf::modified_beta<T>(
190 z.at(i), alpha(energy_keV), beta(energy_keV), z_max(energy_keV));
191 // std::cout << "p(" << z.at(i) << " nm ) = " << p_z.at(i) << " nm^-1\n";
192 weights.at(i) = dz.at(i) * p_z.at(i);
193 slr_rates.at(i) = slr_rate_z<T>(
194 z.at(i), temperature, critical_temperature, gap_meV, xi_0,
195 mean_free_path, lambda_0, exponent, applied_field, dipole_field,
196 correlation_rate, slr_constant, slr_exponent, surface_thickness,
197 surface_rate, electron_phonon_coupling);
198 }
199
200 T sum_weights = std::reduce(std::begin(weights), std::end(weights), 0.0);
201 // std::cout << "sum_weights = " << sum_weights << "\n";
202 T sum_weights_slr = std::transform_reduce(
203 std::begin(weights), std::end(weights), std::begin(slr_rates), 0.0);
204 // std::cout << "sum_weights_slr = " << sum_weights_slr << "\n";
205 T weighted_average = sum_weights_slr / sum_weights;
206 return weighted_average;
207 };
