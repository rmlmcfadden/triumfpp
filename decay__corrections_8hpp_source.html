155double s_31Mg(double time, double beam_rate, double n_31Mg_0) {
156 return ((n_31Mg_0 * lambda_31Mg - beam_rate) *
157 std::exp(-time * lambda_31Mg)) /
158 lambda_31Mg +
159 beam_rate / lambda_31Mg;
160}
161
163double n_31Mg(double time, double beam_pulse, double beam_rate) {
164 if (time > 0 and time <= beam_pulse) {
165 return s_31Mg(time, beam_rate, 0);
166 }
167 if (time > beam_pulse) {
168 double delta = time - beam_pulse;
169 double n_31Mg_0 = s_31Mg(beam_pulse, beam_rate, 0);
170 return s_31Mg(delta, 0, n_31Mg_0);
171 }
172 return 0.0;
173}

176double a_31Mg(double time, double beam_pulse, double beam_rate) {
177 return e_31Mg * lambda_31Mg * n_31Mg(time, beam_pulse, beam_rate);
178}

181double s_31Al(double time, double beam_rate, double n_31Mg_0, double n_31Al_0) {
182 return -((n_31Mg_0 * b_31Mg * lambda_31Mg - beam_rate * b_31Mg) *
183 std::exp(-time * lambda_31Mg)) /
184 (lambda_31Mg - lambda_31Al) +
185 (std::exp(-time * lambda_31Al) *
186 (((n_31Mg_0 * b_31Mg + n_31Al_0) * lambda_31Al - beam_rate * b_31Mg) *
187 lambda_31Mg -
188 n_31Al_0 * std::pow(lambda_31Al, 2))) /
189 (lambda_31Al * lambda_31Mg - std::pow(lambda_31Al, 2)) +
190 (beam_rate * b_31Mg) / lambda_31Al;
191}

194double n_31Al(double time, double beam_pulse, double beam_rate) {
195 if (time > 0 and time <= beam_pulse) {
196 return s_31Al(time, beam_rate, 0, 0);

199 double delta = time - beam_pulse;

201 double n_31Al_0 = s_31Al(beam_pulse, beam_rate, 0, 0);
202 return s_31Al(delta, 0, n_31Mg_0, n_31Al_0);
203 }
204 return 0.0;

213double s_31Si(double time, double beam_rate, double n_31Mg_0, double n_31Al_0,
214 double n_31Si_0) {
215 return ((n_31Si_0 * std::pow(lambda_31Si, 3) +
216 ((-n_31Al_0 * b_31Al - n_31Si_0) * lambda_31Al -
217 n_31Si_0 * lambda_31Mg) *
218 std::pow(lambda_31Si, 2) +
219 (n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al + n_31Si_0) *
220 lambda_31Al * lambda_31Mg * lambda_31Si -
221 beam_rate * b_31Al * b_31Mg * lambda_31Al * lambda_31Mg) *
222 std::exp(-time * lambda_31Si)) /
223 (std::pow(lambda_31Si, 3) +
224 (-lambda_31Mg - lambda_31Al) * std::pow(lambda_31Si, 2) +
225 lambda_31Al * lambda_31Mg * lambda_31Si) -
226 ((n_31Mg_0 * b_31Al * b_31Mg * lambda_31Al * lambda_31Mg -
227 beam_rate * b_31Al * b_31Mg * lambda_31Al) *
228 std::exp(-time * lambda_31Mg)) /
229 ((lambda_31Mg - lambda_31Al) * lambda_31Si -
230 std::pow(lambda_31Mg, 2) + lambda_31Al * lambda_31Mg) +
231 (std::exp(-time * lambda_31Al) *
232 (((n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al) * lambda_31Al -
233 beam_rate * b_31Al * b_31Mg) *
234 lambda_31Mg -
235 n_31Al_0 * b_31Al * std::pow(lambda_31Al, 2))) /
236 ((lambda_31Mg - lambda_31Al) * lambda_31Si -
237 lambda_31Al * lambda_31Mg + std::pow(lambda_31Al, 2)) +
238 (beam_rate * b_31Al * b_31Mg) / lambda_31Si;
239}

262double s_31P(double time, double beam_rate, double n_31Mg_0, double n_31Al_0,
263 double n_31Si_0, double n_31P_0) {
264 return -((n_31Si_0 * std::pow(lambda_31Si, 3) +
265 ((-n_31Al_0 * b_31Al - n_31Si_0) * lambda_31Al -
266 n_31Si_0 * lambda_31Mg) *
267 std::pow(lambda_31Si, 2) +
268 (n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al + n_31Si_0) *
269 lambda_31Al * lambda_31Mg * lambda_31Si -
270 beam_rate * b_31Al * b_31Mg * lambda_31Al * lambda_31Mg) *
271 std::exp(-time * lambda_31Si)) /
272 (std::pow(lambda_31Si, 3) +
273 (-lambda_31Mg - lambda_31Al) * std::pow(lambda_31Si, 2) +
274 lambda_31Al * lambda_31Mg * lambda_31Si) +
275 ((n_31Mg_0 * b_31Al * b_31Mg * lambda_31Al * lambda_31Mg -
276 beam_rate * b_31Al * b_31Mg * lambda_31Al) *
277 std::exp(-time * lambda_31Mg) * lambda_31Si) /
278 ((std::pow(lambda_31Mg, 2) - lambda_31Al * lambda_31Mg) *
279 lambda_31Si -
280 std::pow(lambda_31Mg, 3) +
281 lambda_31Al * std::pow(lambda_31Mg, 2)) -
282 (std::exp(-time * lambda_31Al) *
283 (((n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al) * lambda_31Al -
284 beam_rate * b_31Al * b_31Mg) *
285 lambda_31Mg -
286 n_31Al_0 * b_31Al * std::pow(lambda_31Al, 2)) *
287 lambda_31Si) /
288 ((lambda_31Al * lambda_31Mg - std::pow(lambda_31Al, 2)) *
289 lambda_31Si -
290 std::pow(lambda_31Al, 2) * lambda_31Mg +
291 std::pow(lambda_31Al, 3)) +
292 ((((n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al + n_31Si_0 +
293 n_31P_0) *
294 lambda_31Al -
295 beam_rate * b_31Al * b_31Mg) *
296 lambda_31Mg -
297 beam_rate * b_31Al * b_31Mg * lambda_31Al) *
298 lambda_31Si -
299 beam_rate * b_31Al * b_31Mg * lambda_31Al * lambda_31Mg) /
300 (lambda_31Al * lambda_31Mg * lambda_31Si) +
301 beam_rate * b_31Al * b_31Mg * time;
302}

326double s_30Al(double time, double beam_rate, double n_31Mg_0, double n_30Al_0) {
327 return (((n_31Mg_0 * b_31Mg - n_31Mg_0) * lambda_31Mg - beam_rate * b_31Mg +
328 beam_rate) *
329 std::exp(-time * lambda_31Mg)) /
330 (lambda_31Mg - lambda_30Al) -
331 (std::exp(-time * lambda_30Al) *
332 (((n_31Mg_0 * b_31Mg - n_31Mg_0 - n_30Al_0) * lambda_30Al -
333 beam_rate * b_31Mg + beam_rate) *
334 lambda_31Mg +
335 n_30Al_0 * std::pow(lambda_30Al, 2))) /
336 (lambda_30Al * lambda_31Mg - std::pow(lambda_30Al, 2)) -
337 (beam_rate * b_31Mg - beam_rate) / lambda_30Al;
338}

360double s_30Si(double time, double beam_rate, double n_31Mg_0, double n_31Al_0,
361 double n_30Al_0, double n_30Si_0) {
362 return -((((n_31Mg_0 * b_31Al - n_31Mg_0) * b_31Mg * lambda_31Al +
363 (n_31Mg_0 * b_31Mg - n_31Mg_0) * lambda_30Al) *
364 std::pow(lambda_31Mg, 2) +
365 (((n_31Mg_0 - n_31Mg_0 * b_31Al * b_31Mg) * lambda_30Al +
366 (beam_rate - beam_rate * b_31Al) * b_31Mg) *
367 lambda_31Al +
368 (beam_rate - beam_rate * b_31Mg) * lambda_30Al) *
369 lambda_31Mg +
370 (beam_rate * b_31Al * b_31Mg - beam_rate) * lambda_30Al *
371 lambda_31Al) *
372 std::exp(-time * lambda_31Mg)) /
373 (std::pow(lambda_31Mg, 3) +
374 (-lambda_31Al - lambda_30Al) * std::pow(lambda_31Mg, 2) +
375 lambda_30Al * lambda_31Al * lambda_31Mg) -
376 ((((n_31Mg_0 * b_31Al * b_31Mg + n_31Al_0 * b_31Al - n_31Mg_0 -
377 n_31Al_0 - n_30Si_0 - n_30Al_0) *
378 lambda_30Al -
379 beam_rate * b_31Mg + beam_rate) *
380 lambda_31Al +
381 (beam_rate - beam_rate * b_31Al) * b_31Mg * lambda_30Al) *
382 lambda_31Mg +
383 (beam_rate - beam_rate * b_31Al * b_31Mg) * lambda_30Al *
384 lambda_31Al) /
385 (lambda_30Al * lambda_31Al * lambda_31Mg) +
386 (std::exp(-time * lambda_31Al) *
387 ((((n_31Mg_0 * b_31Al - n_31Mg_0) * b_31Mg + n_31Al_0 * b_31Al -
388 n_31Al_0) *
389 lambda_31Al +
390 (beam_rate - beam_rate * b_31Al) * b_31Mg) *
391 lambda_31Mg +
392 (n_31Al_0 - n_31Al_0 * b_31Al) * std::pow(lambda_31Al, 2))) /
393 (lambda_31Al * lambda_31Mg - std::pow(lambda_31Al, 2)) +
394 (std::exp(-time * lambda_30Al) *
395 (((n_31Mg_0 * b_31Mg - n_31Mg_0 - n_30Al_0) * lambda_30Al -
396 beam_rate * b_31Mg + beam_rate) *
397 lambda_31Mg +
398 n_30Al_0 * std::pow(lambda_30Al, 2))) /
399 (lambda_30Al * lambda_31Mg - std::pow(lambda_30Al, 2)) -
400 beam_rate * b_31Al * b_31Mg * time + beam_rate * time;
401}
