18T pulsed_exp_integral(T time, T time_p, T nuclear_lifetime, T slr_rate) {
19 return nuclear_lifetime * std::exp(-time / nuclear_lifetime) *
20 (std::exp(time_p / nuclear_lifetime + time_p * slr_rate) - 1.0) /
21 ((slr_rate * nuclear_lifetime + 1.0) * std::exp(time * slr_rate));

25template <typename T = double>

27 T slr_rate) {
28 if (time == 0.0) {
29 return asymmetry;
30 } else if (time > 0.0 and time <= pulse_length) {

32 pulsed_exp_integral(time, time, nuclear_lifetime, slr_rate) /
33 normalization(time, nuclear_lifetime);

41 return 0.0;
