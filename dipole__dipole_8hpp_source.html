20// dipole-dipole SLR rate 1/T1 for unlike spins
21// see e.g.,:
22// - M. Mehring, "Principles of High Resolution NMR in Solids", 2nd Ed.
23// (Springer, Berlin, 1983), Ch. 8, p. 262, Eq. (8.21)
24// - H. W. Spiess, "Rotation of molecules and nuclear spin-relaxation", in
25// NMR Basic Principles and Progress Vol. 15, (Springer, Berlin, 1978),
26// p. 121, Tab. 4.6.
27// - A. Abragram, Ch. VIII, p. 295 (but with different numerical factors)?
28template <typename T> T slr_rate(T B_0, T B_d, T nu_c, T gamma_I, T gamma_S) {
29 // NMR frequencies
30 T omega_I = gamma_I * B_0;
31 T omega_S = gamma_S * B_0;
32 // make sure the "coupling" term isn't negative
33 return std::abs(gamma_I * gamma_S) * B_d * B_d *
34 ((1.0 / 3.0) * j<T>(omega_I - omega_S, nu_c) +
35 (1.0 / 1.0) * j(omega_I, nu_c) +
36 (2.0 / 1.0) * j<T>(omega_I + omega_S, nu_c));
37}
